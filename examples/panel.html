packages = [ "bokeh", "numpy", "panel==0.14.1" ] plugins = [ "../build/plugins/python/py_tutor.py" ] import panel as pn slider = pn.widgets.FloatSlider(start=0, end=10, name='Amplitude') def callback(new): return f'Amplitude is: {new}' pn.Row(slider, pn.bind(callback, slider)).servable(target='simple_app');
